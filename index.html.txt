<script>
    let posts = JSON.parse(localStorage.getItem('posts')) || [];

    // Показываем посты при загрузке
    renderPosts();

    function showPage(page) {
        document.getElementById('news-page').style.display = page === 'news' ? 'block' : 'none';
        document.getElementById('messenger').style.display = page === 'messenger' ? 'block' : 'none';
        document.querySelectorAll('nav div').forEach(el => el.classList.remove('active'));
        event.target.classList.add('active');
    }

    function addPost() {
        const text = document.getElementById('postText').value;
        const name = document.getElementById('userName').value || 'Аноним';
        if (!text) return;

        const newPost = {
            id: Date.now(),
            author: name,
            content: text,
            likes: 0,
            liked: false
        };

        posts.unshift(newPost);
        saveAndRender();
        document.getElementById('postText').value = '';
    }

    function toggleLike(id) {
        const post = posts.find(p => p.id === id);
        if (post.liked) {
            post.likes--;
            post.liked = false;
        } else {
            post.likes++;
            post.liked = true;
        }
        saveAndRender();
    }

    function deletePost(id) {
        posts = posts.filter(p => p.id !== id);
        saveAndRender();
    }

    function saveAndRender() {
        localStorage.setItem('posts', JSON.stringify(posts));
        renderPosts();
    }

    function renderPosts() {
        const feed = document.getElementById('feed');
        feed.innerHTML = '';
        posts.forEach(post => {
            feed.innerHTML += `
                <div class="card">
                    <div class="post-header">
                        <div class="avatar" style="background: hsl(${post.id % 360}, 70%, 80%)"></div>
                        ${post.author}
                        <span class="del-btn" onclick="deletePost(${post.id})">Удалить</span>
                    </div>
                    <div class="post-content">${post.content}</div>
                    <div class="post-footer">
                        <div class="like-btn ${post.liked ? 'active' : ''}" onclick="toggleLike(${post.id})">
                            ❤ ${post.likes}
                        </div>
                    </div>
                </div>
            `;
        });
    }

    function sendMsg() {
        const input = document.getElementById('msgInput');
        if (!input.value) return;
        const chat = document.getElementById('chatBox');
        chat.innerHTML += `<div class="msg" style="align-self: flex-end; background: #cee3ff;">${input.value}</div>`;
        input.value = '';
        chat.scrollTop = chat.scrollHeight;
    }
</script>
